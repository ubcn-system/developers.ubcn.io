    ### API 호출 방법
    API 호출 방법은 다음과 같습니다.
```ts

const API = axios.create({
  baseURL: `http://api.ubcn.co.kr/api`,
  headers: {
    "Access-Control-Allow-Origin": "*",
    "company": "{company}",
    "token": "{token}",
  },
});

//get, post 모두 지원합니다.
const res = await API.get("{service}");
console.log(res.data);

// 프로세스가 제대로 완료되지 않은 경우 에러 코드가 존재합니다.
if (res.code != null) {
  return alert(res.decription);
}

```
